00285 // Returns true if a vertex or any of its parents is on a boundary
00286 bool VertexOnBoundary( xyzvertex const * v ) {
00287 
00288     if (not v)
00289         return false;
00290 
00291     if (v->OnBoundary())
00292         return true;
00293 
00294     xyzvertex const * pv = v->GetParentVertex();
00295     if (pv)
00296         return VertexOnBoundary(pv);
00297     else {
00298         xyzhalfedge const * pe = v->GetParentEdge();
00299         if (pe) {
00300               return VertexOnBoundary(pe->GetOrgVertex()) or
00301                      VertexOnBoundary(pe->GetDestVertex());
00302         } else {
00303             xyzface const * pf = v->GetParentFace(), * rootf = pf;
00304             while (pf) {
00305                 pf = pf->GetParent();
00306                 if (pf)
00307                     rootf=pf;
00308             }
00309             if (rootf)
00310                 for (int i=0; i<rootf->GetNumVertices(); ++i)
00311                     if (rootf->GetVertex(i)->OnBoundary())
00312                         return true;
00313         }
00314     }
00315     return false;
00316 }

00319 int checkMesh( char const * msg, xyzmesh * hmesh, int levels, Scheme scheme=kCatmark ) {
00320 
00321     assert(msg);
00322 
00323     int count=0;
00324     float deltaAvg[3] = {0.0f, 0.0f, 0.0f},
00325           deltaCnt[3] = {0.0f, 0.0f, 0.0f};
00326 
00327     fMeshFactory fact( hmesh, levels );
00328     fMesh * m = fact.Create( );
00329     m->Subdivide( );
00330 
00331     if (g_debugmode) {
00332         for (int i=1; i<=levels; ++i)
00333             if (g_dumphbr)
00334                 dumpXYZMesh( hmesh, i, scheme );
00335             else
00336                 dumpMesh( m, i, scheme );
00337     } else
00338         printf("- %s (scheme=%d)\n", msg, scheme);
00339 
00340     std::vector<int> const & remap = fact.GetRemappingTable();
00341 
00342     int nverts = m->GetNumVertices();
00343 
00344     // compare vertex results (only position for now - we need to expand w/ some vertex data)
00345     for (int i=1; i<nverts; ++i) {
00346 
00347         xyzvertex * hv = hmesh->GetVertex(i);
00348         xyzVV & nv = m->GetVertex( remap[hv->GetID()] );
00349 
00350         // boundary interpolation rules set to "none" produce "undefined" vertices on
00351         // boundary vertices : far does not match hbr for those, so skip comparison.
00352         if ( hmesh->GetInterpolateBoundaryMethod()==xyzmesh::k_InterpolateBoundaryNone and
00353              VertexOnBoundary(hv) )
00354              continue;
00355 
00356 
00357         if ( hv->GetData().GetPos()[0] != nv.GetPos()[0] )
00358             deltaCnt[0]++;
00359         if ( hv->GetData().GetPos()[1] != nv.GetPos()[1] )
00360             deltaCnt[1]++;
00361         if ( hv->GetData().GetPos()[2] != nv.GetPos()[2] )
00362             deltaCnt[2]++;
00363 
00364         float delta[3] = { hv->GetData().GetPos()[0] - nv.GetPos()[0],
00365                            hv->GetData().GetPos()[1] - nv.GetPos()[1],
00366                            hv->GetData().GetPos()[2] - nv.GetPos()[2] };
00367 
00368         deltaAvg[0]+=delta[0];
00369         deltaAvg[1]+=delta[1];
00370         deltaAvg[2]+=delta[2];
00371 
00372         float dist = sqrtf( delta[0]*delta[0]+delta[1]*delta[1]+delta[2]*delta[2]);
00373         if ( dist > PRECISION ) {
00374             if (not g_debugmode)
00375                 printf("// HbrVertex<T> %d fails : dist=%.10f (%.10f %.10f %.10f)"
00376                        " (%.10f %.10f %.10f)\n", i, dist, hv->GetData().GetPos()[0],
00377                                                           hv->GetData().GetPos()[1],
00378                                                           hv->GetData().GetPos()[2],
00379                                                           nv.GetPos()[0],
00380                                                           nv.GetPos()[1],
00381                                                           nv.GetPos()[2] );
00382            count++;
00383         }
00384     }
00385 
00386     if (deltaCnt[0])
00387         deltaAvg[0]/=deltaCnt[0];
00388     if (deltaCnt[1])
00389         deltaAvg[1]/=deltaCnt[1];
00390     if (deltaCnt[2])
00391         deltaAvg[2]/=deltaCnt[2];
00392 
00393     if (not g_debugmode) {
00394         printf("  delta ratio : (%d/%d %d/%d %d/%d)\n", (int)deltaCnt[0], nverts,
00395                                                         (int)deltaCnt[1], nverts,
00396                                                         (int)deltaCnt[2], nverts );
00397         printf("  average delta : (%.10f %.10f %.10f)\n", deltaAvg[0],
00398                                                           deltaAvg[1],
00399                                                           deltaAvg[2] );
00400         if (count==0)
00401             printf("  success !\n");
00402     }
00403 
00404     delete hmesh;
00405     delete m;
00406 
00407     return count;
00408 }
