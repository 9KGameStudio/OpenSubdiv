00606 template <class T> void
00607 createTopology( shape const * sh, OpenSubdiv::HbrMesh<T> * mesh, Scheme scheme) {
00608 
00609       const int * fv=&(sh->faceverts[0]);
00610       for(int f=0, ptxidx=0;f<sh->getNfaces(); f++ ) {
00611 
00612           int nv = sh->nvertsPerFace[f];
00613 
00614           if ((scheme==kLoop) and (nv!=3)) {
00615               printf("Trying to create a Loop surbd with non-triangle face\n");
00616               exit(1);
00617           }
00618 
00619           for(int j=0;j<nv;j++) {
00620               OpenSubdiv::HbrVertex<T> * origin      = mesh->GetVertex( fv[j] );
00621               OpenSubdiv::HbrVertex<T> * destination = mesh->GetVertex( fv[ (j+1)%nv] );
00622               OpenSubdiv::HbrHalfedge<T> * opposite  = destination->GetEdge(origin);
00623 
00624               if(origin==NULL || destination==NULL) {
00625                   printf(" An edge was specified that connected a nonexistent vertex\n");
00626                   exit(1);
00627               }
00628 
00629               if(origin == destination) {
00630                   printf(" An edge was specified that connected a vertex to itself\n");
00631                   exit(1);
00632               }
00633 
00634               if(opposite && opposite->GetOpposite() ) {
00635                   printf(" A non-manifold edge incident to more than 2 faces was found\n");
00636                   exit(1);
00637               }
00638 
00639               if(origin->GetEdge(destination)) {
00640                   printf(" An edge connecting two vertices was specified more than once."
00641                          " It's likely that an incident face was flipped\n");
00642                   exit(1);
00643               }
00644           }
00645 
00646           OpenSubdiv::HbrFace<T> * face = mesh->NewFace(nv, (int *)fv, 0);
00647 
00648           face->SetPtexIndex(ptxidx);
00649 
00650           if ( (scheme==kCatmark or scheme==kBilinear) and nv != 4 )
00651               ptxidx+=nv;
00652           else
00653               ptxidx++;
00654 
00655           fv+=nv;
00656       }
00657 
00658       mesh->SetInterpolateBoundaryMethod( OpenSubdiv::HbrMesh<T>::k_InterpolateBoundaryEdgeOnly );
00659 
00660       applyTags<T>( mesh, sh );
00661 
00662       mesh->Finish();
00663 }
