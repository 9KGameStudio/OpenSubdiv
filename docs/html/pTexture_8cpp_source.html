00106         int darea = a->current.ulog2 * a->current.vlog2  -

00409 // attempt to re-size per-face resolutions to hit the uncompressed texel
00410 // memory use requirement
00411 void
00412 loader::OptimizeResolution( unsigned long int memrec )
00413 {
00414     unsigned long int txrec = memrec / _bpp;
00415 
00416     if (txrec==_txc)
00417         return;
00418     else
00419     {
00420         unsigned long int txcur = _txc;
00421 
00422         if (_blocks.size()==0)
00423             return;
00424 
00425         std::vector<block *> blocks( _blocks.size() );
00426         for (unsigned long int i=0; i<blocks.size(); ++i)
00427             blocks[i] = &(_blocks[i]);
00428 
00429         // reducing footprint ----------------------------------------
00430         if (txrec < _txc)
00431         {
00432             // blocks that have already been resized heavily will be considered last
00433             std::sort(blocks.begin(), blocks.end(), block::downsizePredicate );
00434 
00435             while ( (txcur>0) && (txcur>txrec) )
00436             {
00437                 unsigned long int txsaved = txcur;
00438 
00439                 // start stealing from largest to smallest down
00440                 for (int i=(int)blocks.size()-1; i>=0; --i)
00441                 {
00442                     block * b = blocks[i];
00443 
00444                     // we have already hit rock bottom resolution... skip this block
00445                     if (b->current.ulog2==0 || b->current.vlog2==0)
00446                          continue;
00447 
00448                     GLushort ures = (1<<(unsigned)(b->current.ulog2-1)),
00449                              vres = (1<<(unsigned)(b->current.vlog2-1));
00450 
00451                     int diff = b->current.size() - ures * vres;
00452 
00453                     // we are about to overshoot the limit with our big blocks :
00454                     // skip until we find something smaller
00455                     if ( ((unsigned long int)diff>txcur) || ((txcur-diff)<txrec) )
00456                         continue;
00457 
00458                     b->current.ulog2--;
00459                     b->current.vlog2--;
00460                     txcur-=diff;
00461                 }
00462 
00463                 // couldn't scavenge anymore even from smallest faces : time to bail out.
00464                 if (txsaved==txcur)
00465                     break;
00466             }
00467             _txc = txcur;
00468         } else {
00469 
00470         // increasing footprint --------------------------------------
00471 
00472             // blocks that have already been resized heavily will be considered first
00473             std::sort(blocks.begin(), blocks.end(), block::upsizePredicate );
00474 
00475             while ( (txcur < _txn) && (txcur < txrec) )
00476             {
00477                 unsigned long int txsaved = txcur;
00478 
00479                 // start adding back to the largest faces first
00480                 for (int i=0; i<(int)blocks.size(); ++i)
00481                 {
00482                     block * b = blocks[i];
00483 
00484                     // already at native resolution... nothing to be done
00485                     if (b->current == b->native)
00486                         continue;
00487 
00488                     GLushort ures = (1<<(unsigned)(b->current.ulog2+1)),
00489                              vres = (1<<(unsigned)(b->current.vlog2+1));
00490 
00491                     int diff = ures * vres - b->current.size();
00492 
00493                     // we are about to overshoot the limit with our big blocks :
00494                     // skip until we find something smaller
00495                     if ( (txcur + diff) > txrec )
00496                         continue;
00497 
00498                     b->current.ulog2++;
00499                     b->current.vlog2++;
00500                     txcur+=diff;
00501                 }
00502 
00503                 // couldn't scavenge anymore even from smallest faces : time to bail out.
00504                 if (txsaved==txcur)
00505                     break;
00506             }
00507             _txc = txcur;
00508         }
00509     }

00512 // greedy packing of blocks into pages
00513 void
00514 loader::OptimizePacking( int maxnumpages )
00515 {
00516     if (_blocks.size()==0)
00517         return;
00518 
00519     // generate a vector of pointers to the blocks -------------------
00520     std::vector<block *> blocks( _blocks.size() );
00521     for (unsigned long int i=0; i<blocks.size(); ++i)
00522         blocks[i] = &(_blocks[i]);
00523 
00524     // intro-sort blocks from largest to smallest (helps minimizing waste with
00525     // greedy packing)
00526     std::sort(blocks.rbegin(), blocks.rend(), block::currentAreaSort );
00527 
00528     // compute page size ---------------------------------------------
00529     // page size is set to the largest edge of the largest block : this is the
00530     // smallest possible page size, which should minimize the texels wasted on
00531     // the "last page" when the smallest blocks are being packed.
00532     _pagesize = blocks[0]->current.ulog2 > blocks[0]->current.vlog2 ?
00533                 blocks[0]->current.u() : blocks[0]->current.v();
00534 
00535     // at least 2*GUTTER_WIDTH of margin required for each page to fit
00536     _pagesize += OsdPTexture::GetPageMargin();
00537 
00538     // grow the pagesize to make sure the optimization will not exceed the maximum
00539     // number of pages allowed
00540     for (int npages=_txc/(_pagesize*_pagesize); npages>maxnumpages; _pagesize<<=1)
00541         npages = _txc/(_pagesize*_pagesize );
00542 
00543     ClearPages( );
00544 
00545     // save some memory allocation time : guess the number of pages from the
00546     // number of texels
00547     _pages.reserve( _txc / (_pagesize*_pagesize) + 1 );
00548 
00549     // pack blocks into slots ----------------------------------------
00550     for (unsigned long int i=0, firstslot=0; i<_blocks.size(); ++i ) {
00551 
00552         block * b = blocks[i];
00553 
00554         // traverse existing pages for a suitable slot ---------------
00555         bool added=false;
00556         for( unsigned long int p=firstslot; p<_pages.size(); ++p )
00557             if( (added=_pages[p]->addBlock( b )) ) {
00558                 break;
00559             }
00560 
00561         // if none was found : start new page
00562         if( !added ) {
00563             page * p = new page( _pagesize );
00564             p->addBlock(b);
00565             _pages.push_back( p );
00566         }
00567 
00568         // adjust the page flag to the first page with open slots
00569         if( (_pages.size()>(firstslot+1)) &&
00570             (_pages[firstslot+1]->isFull()) )
00571             ++firstslot;
00572     }
00573 }

00641 // sample neighbor face's edge
00642 static void
00643 sampleNeighbor(PtexTexture * ptex, GLubyte *border, int face, int edge, int length, int bpp)
00644 {
00645     const Ptex::FaceInfo &fi = ptex->getFaceInfo(face);
00646 
00647     // copy adjacent borders
00648     int adjface = fi.adjface(edge);
00649     if(adjface != -1) {
00650         int ae = fi.adjedge(edge);
00651         if (!fi.isSubface() && ptex->getFaceInfo(adjface).isSubface()) {
00652             /* nonsubface -> subface (1:0.5)  see http://ptex.us/adjdata.html for more detail
00653               +------------------+
00654               |       face       |
00655               +--------edge------+
00656               | adj face |       |
00657               +----------+-------+
00658             */
00659             resampleBorder(ptex, adjface, ae, border, edge, length/2, bpp);
00660             const Ptex::FaceInfo &sfi1 = ptex->getFaceInfo(adjface);
00661             adjface = sfi1.adjface((ae+3)%4);
00662             const Ptex::FaceInfo &sfi2 = ptex->getFaceInfo(adjface);
00663             ae = (sfi1.adjedge((ae+3)%4)+3)%4;
00664             resampleBorder(ptex, adjface, ae, border+(length/2*bpp), edge, length/2, bpp);
00665 
00666         } else if (fi.isSubface() && !ptex->getFaceInfo(adjface).isSubface()) {
00667             /* subface -> nonsubface (0.5:1).   two possible configuration
00668                      case 1                    case 2
00669               +----------+----------+  +----------+----------+--------+
00670               |   face   |    B     |  |          |  face    |   B    |
00671               +---edge---+----------+  +----------+--edge----+--------+
00672               |0.0      0.5      1.0|  |0.0      0.5      1.0|
00673               |       adj face      |  |       adj face      |
00674               +---------------------+  +---------------------+
00675             */
00676             int Bf = fi.adjface((edge+1)%4);
00677             int Be = fi.adjedge((edge+1)%4);
00678             int f = ptex->getFaceInfo(Bf).adjface((Be+1)%4);
00679             int e = ptex->getFaceInfo(Bf).adjedge((Be+1)%4);
00680             if(f == adjface && e == ae) // case 1
00681                 resampleBorder(ptex, adjface, ae, border, edge, length, bpp, 0.0, 0.5);
00682             else  // case 2
00683                 resampleBorder(ptex, adjface, ae, border, edge, length, bpp, 0.5, 1.0);
00684 
00685         } else {
00686             /*  ordinary case (1:1 match)
00687                 +------------------+
00688                 |       face       |
00689                 +--------edge------+
00690                 |    adj face      |
00691                 +----------+-------+
00692             */
00693             resampleBorder(ptex, adjface, ae, border, edge, length, bpp);
00694         }
00695     } else {
00696         /* border edge. duplicate itself
00697            +-----------------+
00698            |       face      |
00699            +-------edge------+
00700         */
00701         resampleBorder(ptex, face, edge, border, edge, length, bpp);
00702         flipBuffer(border, length, bpp);
00703     }
00704 }

00706 // average corner pixels by traversing all adjacent faces around vertex
00707 //
00708 static bool
00709 averageCorner(PtexTexture *ptex, float *accumPixel, int numchannels, int face, int edge)
00710 {
00711     const Ptex::FaceInfo &fi = ptex->getFaceInfo(face);
00712 
00713     int adjface = fi.adjface(edge);
00714 
00715     // don't average T-vertex.
00716     if (fi.isSubface() && !ptex->getFaceInfo(adjface).isSubface())
00717         return false;
00718 
00719     int valence = 0;
00720     int currentFace = face;
00721     int currentEdge = edge;
00722     int uv[4][2] = {{0,0}, {1,0}, {1,1}, {0,1}};
00723     float *pixel = (float*)alloca(sizeof(float)*numchannels);
00724 
00725     // clear result buffer
00726     memset(accumPixel, 0, sizeof(float)*numchannels);
00727 
00728     do {
00729         valence++;
00730         Ptex::FaceInfo info = ptex->getFaceInfo(currentFace);
00731         ptex->getPixel(currentFace,
00732                         uv[currentEdge][0] * (info.res.u()-1),
00733                         uv[currentEdge][1] * (info.res.v()-1),
00734                         pixel, 0, numchannels);
00735         for(int j=0; j<numchannels; ++j) {
00736             accumPixel[j] += pixel[j];
00737         }
00738 
00739         // next face
00740         currentFace = info.adjface(currentEdge);
00741         currentEdge = info.adjedge(currentEdge);
00742         currentEdge = (currentEdge+1)%4;
00743     } while(currentFace != -1 && currentFace != face);
00744 
00745     for(int j=0; j<numchannels; ++j) {
00746         accumPixel[j] /= valence;
00747     }
00748 
00749     return true;
00750 }

00752 // sample neighbor pixels and populate around blocks
00753 static void
00754 guttering(PtexTexture *_ptex, block *b, GLubyte *pptr, int _bpp, int _pagesize, int stride, int gwidth)
00755 {
00756     const Ptex::FaceInfo &fi = _ptex->getFaceInfo(b->idx);
00757     GLubyte * border = new GLubyte[_pagesize * _bpp];
00758 
00759     for(int w=0; w<gwidth; ++w) {
00760         for(int edge=0; edge<4; edge++) {
00761 
00762             int len = (edge==0 or edge==2) ? b->current.u() : b->current.v();
00763             // XXX: for now, sample same edge regardless of gutter depth
00764             sampleNeighbor(_ptex, border, b->idx, edge, len, _bpp);
00765 
00766             GLubyte *s = border, *d;
00767             for(int j=0;j<len;++j) {
00768                 d = pptr;
00769                 switch(edge) {
00770                 case Ptex::e_bottom:
00771                     d += stride*(b->v-1-w) + _bpp*(b->u+j);
00772                     break;
00773                 case Ptex::e_right:
00774                     d += stride*(b->v+j) + _bpp*(b->u+b->current.u()+w);
00775                     break;
00776                 case Ptex::e_top:
00777                     d += stride*(b->v+b->current.v()+w) + _bpp*(b->u+len-j-1);
00778                     break;
00779                 case Ptex::e_left:
00780                     d += stride*(b->v+len-j-1) + _bpp*(b->u-1-w);
00781                     break;
00782                 }
00783                 for(int k=0; k<_bpp; k++)
00784                     *d++ = *s++;
00785             }
00786         }
00787     }
00788     delete[] border;
00789 
00790     // average corner pixels
00791     int numchannels = _ptex->numChannels();
00792     float *accumPixel = new float[numchannels];
00793     int uv[4][2] = {{-1,-1}, {1,-1}, {1,1}, {-1,1}};
00794     for(int edge=0; edge<4; edge++) {
00795 
00796         if(averageCorner(_ptex, accumPixel, numchannels, b->idx, edge)) {
00797             // set accumPixel to 4 corner
00798             int du = (b->u+gwidth*uv[edge][0]);
00799             int dv = (b->v+gwidth*uv[edge][1]);
00800             if(edge==1||edge==2) du += b->current.u()-gwidth-1;
00801             if(edge==2||edge==3) dv += b->current.v()-gwidth-1;
00802             // .. over (gwidth+1)x(gwidth+1) pixels for each corner
00803             for(int u=0; u<=gwidth; ++u) {
00804                 for(int v=0; v<=gwidth; ++v) {
00805                     GLubyte *d = pptr + (dv+u)*stride + (du+v)*_bpp;
00806                     Ptex::ConvertFromFloat(d, accumPixel, _ptex->dataType(), numchannels);
00807                 }
00808             }
00809         }
00810     }
00811     delete[] accumPixel;
00812 }

00815 // PTex texels
00816 bool
00817 loader::GenerateBuffers( )
00818 {
00819     if (_pages.size()==0) return false;
00820 
00821     // populate the page index lookup texture ------------------------
00822     _indexBuffer = new GLuint[ _blocks.size() ];

00824         page * p = _pages[i];
00825         for (page::blist::iterator j=p->blocks.begin(); j!=p->blocks.end(); ++j)
00826             _indexBuffer[ (*j)->idx ] = i;
00827     }
00828 
00829     // populate the layout lookup texture ----------------------------
00830     GLfloat * lptr = _layoutBuffer = new GLfloat[ 4 * _blocks.size() ];
00831     for (unsigned long int i=0; i<_blocks.size(); ++ i) {
00832         // normalize coordinates by pagesize resolution !
00833         *lptr++ = (GLfloat) _blocks[i].u / (GLfloat) _pagesize;
00834         *lptr++ = (GLfloat) _blocks[i].v / (GLfloat) _pagesize;
00835         *lptr++ = (GLfloat) _blocks[i].current.u() / (GLfloat) _pagesize;
00836         *lptr++ = (GLfloat) _blocks[i].current.v() / (GLfloat) _pagesize;
00837     }
00838 
00839     // populate the texels -------------------------------------------
00840     int stride = _bpp * _pagesize,
00841         pagestride = stride * _pagesize;
00842 
00843     GLubyte * pptr = _texelBuffer = new GLubyte[ pagestride * _pages.size() ];
00844 
00845     for (unsigned long int i=0; i<_pages.size(); i++) {
00846 
00847         page * p = _pages[i];
00848 
00849         for (page::blist::iterator b=p->blocks.begin(); b!=p->blocks.end(); ++b) {
00850             _ptex->getData( (*b)->idx, pptr + stride*(*b)->v + _bpp*(*b)->u, stride, (*b)->current );
00851 
00852             if(OsdPTexture::GetGutterWidth() > 0)
00853                 guttering(_ptex, *b, pptr, _bpp, _pagesize, stride, OsdPTexture::GetGutterWidth());
