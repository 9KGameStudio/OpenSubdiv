00267 inline void
00268 cross(float *n, const float *p0, const float *p1, const float *p2) {
00269 
00270     float a[3] = { p1[0]-p0[0], p1[1]-p0[1], p1[2]-p0[2] };
00271     float b[3] = { p2[0]-p0[0], p2[1]-p0[1], p2[2]-p0[2] };
00272     n[0] = a[1]*b[2]-a[2]*b[1];
00273     n[1] = a[2]*b[0]-a[0]*b[2];
00274     n[2] = a[0]*b[1]-a[1]*b[0];
00275 
00276     float rn = 1.0f/sqrtf(n[0]*n[0] + n[1]*n[1] + n[2]*n[2]);
00277     n[0] *= rn;
00278     n[1] *= rn;
00279     n[2] *= rn;
00280 }
