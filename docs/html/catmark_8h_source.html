00625 template <class T>
00626 void
00627 HbrCatmarkSubdivision<T>::GuaranteeNeighbor(HbrMesh<T>* mesh, HbrHalfedge<T>* edge) {
00628     if (edge->GetOpposite()) {
00629         return;
00630     }
00631 
00632     // For the given edge: if the parent of either of its incident
00633     // vertices is itself a _face_, then ensuring that this parent
00634     // face has refined at a particular vertex is sufficient to
00635     // ensure that both of the faces on each side of the edge have
00636     // been created.
00637     bool destParentWasEdge = true;
00638     HbrFace<T>* parentFace = edge->GetOrgVertex()->GetParentFace();
00639     HbrHalfedge<T>* parentEdge = edge->GetDestVertex()->GetParentEdge();
00640     if (!parentFace) {
00641         destParentWasEdge = false;
00642         parentFace = edge->GetDestVertex()->GetParentFace();
00643         parentEdge = edge->GetOrgVertex()->GetParentEdge();
00644     }
00645 
00646     if (parentFace) {
00647 
00648         // Make sure we deal with a parent halfedge which is
00649         // associated with the parent face
00650         if (parentEdge->GetFace() != parentFace) {
00651             parentEdge = parentEdge->GetOpposite();
00652         }
00653         // If one of the vertices had a parent face, the other one MUST
00654         // have been a child of an edge
00655         assert(parentEdge && parentEdge->GetFace() == parentFace);
00656 #ifdef HBR_DEBUG
00657         std::cerr << "\nparent edge is " << *parentEdge << "\n";
00658 #endif
00659 
00660         // The vertex to refine at depends on whether the
00661         // destination or origin vertex of this edge had a parent
00662         // edge
00663         if (destParentWasEdge) {
00664             RefineFaceAtVertex(mesh, parentFace, parentEdge->GetOrgVertex());
00665         } else {
00666             RefineFaceAtVertex(mesh, parentFace, parentEdge->GetDestVertex());
00667         }
00668 
00669         // It should always be the case that the opposite now exists -
00670         // we can't have a boundary case here
00671         assert(edge->GetOpposite());
00672     } else {
00673         HbrVertex<T>* parentVertex = edge->GetOrgVertex()->GetParentVertex();
00674         parentEdge = edge->GetDestVertex()->GetParentEdge();
00675         if (!parentVertex) {
00676             parentVertex = edge->GetDestVertex()->GetParentVertex();
00677             parentEdge = edge->GetOrgVertex()->GetParentEdge();
00678         }
00679 
00680         if (parentVertex) {
00681 
00682             assert(parentEdge);
00683 
00684 #ifdef HBR_DEBUG
00685             std::cerr << "\nparent edge is " << *parentEdge << "\n";
00686 #endif
00687 
00688             // 1. Go up to the parent of my face
00689 
00690             parentFace = edge->GetFace()->GetParent();
00691 #ifdef HBR_DEBUG
00692             std::cerr << "\nparent face is " << *parentFace << "\n";
00693 #endif
00694 
00695             // 2. Ask the opposite face (if it exists) to refine
00696             if (parentFace) {
00697 
00698                 // A vertex can be associated with either of two
00699                 // parent halfedges. If the parent edge that we're
00700                 // interested in doesn't match then we should look at
00701                 // its opposite
00702                 if (parentEdge->GetFace() != parentFace)
00703                     parentEdge = parentEdge->GetOpposite();
00704                 assert(parentEdge->GetFace() == parentFace);
00705 
00706                 // Make sure the parent edge has its neighbor as well
00707                 GuaranteeNeighbor(mesh, parentEdge);
00708 
00709                 // Now access that neighbor and refine it
00710                 if (parentEdge->GetRightFace()) {
00711                     RefineFaceAtVertex(mesh, parentEdge->GetRightFace(), parentVertex);
00712 
00713                     // FIXME: assertion?
00714                     assert(edge->GetOpposite());
00715                 }
00716             }
00717         }
00718     }

00721 template <class T>
00722 void
00723 HbrCatmarkSubdivision<T>::GuaranteeNeighbors(HbrMesh<T>* mesh, HbrVertex<T>* vertex) {
00724 
00725 #ifdef HBR_DEBUG
00726     std::cerr << "\n\nneighbor guarantee at " << *vertex << " invoked\n";
00727 #endif
00728 
00729     // If the vertex is a child of a face, guaranteeing the neighbors
00730     // of the vertex is simply a matter of ensuring the parent face
00731     // has refined.
00732     HbrFace<T>* parentFace = vertex->GetParentFace();
00733     if (parentFace) {
00734 
00735 #ifdef HBR_DEBUG
00736         std::cerr << "  forcing full refine on parent face\n";
00737 #endif
00738         Refine(mesh, parentFace);
00739         return;
00740     }
00741 
00742     // Otherwise if the vertex is a child of an edge, we need to
00743     // ensure that the parent faces on either side of the parent edge
00744     // 1) exist, and 2) have refined at both vertices of the parent
00745     // edge
00746     HbrHalfedge<T>* parentEdge = vertex->GetParentEdge();
00747     if (parentEdge) {
00748 
00749 #ifdef HBR_DEBUG
00750         std::cerr << "  forcing full refine on adjacent faces of parent edge\n";
00751 #endif
00752         HbrVertex<T>* dest = parentEdge->GetDestVertex();
00753         HbrVertex<T>* org = parentEdge->GetOrgVertex();
00754         GuaranteeNeighbor(mesh, parentEdge);
00755         parentFace = parentEdge->GetLeftFace();
00756         RefineFaceAtVertex(mesh, parentFace, dest);
00757         RefineFaceAtVertex(mesh, parentFace, org);
00758 
00759 #ifdef HBR_DEBUG
00760         std::cerr << "    on the right face?\n";
00761 #endif
00762         parentFace = parentEdge->GetRightFace();
00763         // The right face may not necessarily exist even after
00764         // GuaranteeNeighbor
00765         if (parentFace) {
00766             RefineFaceAtVertex(mesh, parentFace, dest);
00767             RefineFaceAtVertex(mesh, parentFace, org);
00768         }
00769 #ifdef HBR_DEBUG
00770         std::cerr << "  end force\n";
00771 #endif
00772         return;
00773     }
00774 
00775     // The last case: the vertex is a child of a vertex. In this case
00776     // we have to first recursively guarantee that the parent's
00777     // adjacent faces also exist.
00778     HbrVertex<T>* parentVertex = vertex->GetParentVertex();
00779     if (parentVertex) {
00780 
00781 #ifdef HBR_DEBUG
00782         std::cerr << "  recursive parent vertex guarantee call\n";
00783 #endif
00784         parentVertex->GuaranteeNeighbors();
00785 
00786         // And then we refine all the face neighbors of the
00787         // parentVertex
00788         HbrHalfedge<T>* start = parentVertex->GetIncidentEdge(), *edge;
00789         edge = start;
00790         while (edge) {
00791             HbrFace<T>* f = edge->GetLeftFace();
00792             RefineFaceAtVertex(mesh, f, parentVertex);
00793             edge = parentVertex->GetNextEdge(edge);
00794             if (edge == start) break;
00795         }
00796     }
00797 }

01090     }
01091     for (int i = 0; i < passes; ++i) {
01092         switch (masks[i]) {
01093             case HbrVertex<T>::k_Smooth:
01094             case HbrVertex<T>::k_Dart: {
01095                 // Compute n-2/n of the old vertex value
01096                 data.AddWithWeight(vertex->GetData(), weights[i] * invvalencesquared * valence * (valence - 2));
01097                 // Add 1 / n^2 * surrounding edge vertices and surrounding face
01098                 // subdivided vertices
01099                 HbrSubdivision<T>::AddSurroundingVerticesWithWeight(
01100                     mesh, vertex, weights[i] * invvalencesquared, &data);
01101 
01102                 HbrHalfedge<T>* start = vertex->GetIncidentEdge(), *edge;
01103                 edge = start;
01104                 while (edge) {
01105                     HbrFace<T>* f = edge->GetLeftFace();

01107                     edge = vertex->GetNextEdge(edge);
01108                     if (edge == start) break;
01109                 }
01110                 break;
01111             }
01112             case HbrVertex<T>::k_Crease: {
01113                 // Compute 3/4 of old vertex value
01114                 data.AddWithWeight(vertex->GetData(), weights[i] * 0.75f);
01115 
01116                 // Add 0.125f of the (hopefully only two!) neighbouring
01117                 // sharp edges
01118                 HbrSubdivision<T>::AddCreaseEdgesWithWeight(
01119                     mesh, vertex, i == 1, weights[i] * 0.125f, &data);
01120                 break;
01121             }
01122             case HbrVertex<T>::k_Corner:
01123             default: {
01124                 // Just copy the old value
01125                 data.AddWithWeight(vertex->GetData(), weights[i]);
01126                 break;
01127             }
01128         }
01129     }
01130 
01131     // Varying data is always just propagated down
01132     data.AddVaryingWithWeight(vertex->GetData(), 1.0f);
01133 
01134 #ifdef HBR_DEBUG
01135     std::cerr << "Subdividing at " << *vertex << "\n";
01136     std::cerr << "  created " << *v << "\n";
01137 #endif
01138     // Inherit extraordinary flag and sharpness
01139     if (vertex->IsExtraordinary()) v->SetExtraordinary();
01140     float sharp = vertex->GetSharpness();
01141     if (sharp >= HbrVertex<T>::k_InfinitelySharp) {
01142         v->SetSharpness(HbrVertex<T>::k_InfinitelySharp);
01143     } else if (sharp > HbrVertex<T>::k_Smooth) {
01144         sharp -= 1.0f;
01145         if (sharp < (float) HbrVertex<T>::k_Smooth) {
01146             sharp = (float) HbrVertex<T>::k_Smooth;
01147         }
01148         v->SetSharpness(sharp);
01149     } else {
01150         v->SetSharpness(HbrVertex<T>::k_Smooth);
01151     }
