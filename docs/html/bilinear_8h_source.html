00578                 }

00581                 // sure the children get their indices reassigned to be

00584                     child->SetPtexIndex(face->GetPtexIndex() + i);
00585                 }
00586 
00587                 transferEditsToChild(face, child, i);
00588                 return child;
00589             } else {
00590                 return face->GetChild(i);

00592         }
00593         prevedge = edge;

00600 void
00601 HbrBilinearSubdivision<T>::GuaranteeNeighbor(HbrMesh<T>* mesh, HbrHalfedge<T>* edge) {
00602     if (edge->GetOpposite()) {
00603         return;
00604     }
00605 
00606     // For the given edge: if the parent of either of its incident
00607     // vertices is itself a _face_, then ensuring that this parent
00608     // face has refined at a particular vertex is sufficient to
00609     // ensure that both of the faces on each side of the edge have
00610     // been created.
00611     bool destParentWasEdge = true;
00612     HbrFace<T>* parentFace = edge->GetOrgVertex()->GetParentFace();
00613     HbrHalfedge<T>* parentEdge = edge->GetDestVertex()->GetParentEdge();
00614     if (!parentFace) {
00615         destParentWasEdge = false;
00616         parentFace = edge->GetDestVertex()->GetParentFace();
00617         parentEdge = edge->GetOrgVertex()->GetParentEdge();
00618     }
00619 
00620     if (parentFace) {
00621 
00622         // Make sure we deal with a parent halfedge which is
00623         // associated with the parent face
00624         if (parentEdge->GetFace() != parentFace) {
00625             parentEdge = parentEdge->GetOpposite();
00626         }
00627         // If one of the vertices had a parent face, the other one MUST
00628         // have been a child of an edge
00629         assert(parentEdge && parentEdge->GetFace() == parentFace);
00630 #ifdef HBR_DEBUG
00631         std::cerr << "\nparent edge is " << *parentEdge << "\n";
00632 #endif
00633 
00634         // The vertex to refine at depends on whether the
00635         // destination or origin vertex of this edge had a parent
00636         // edge
00637         if (destParentWasEdge) {
00638             RefineFaceAtVertex(mesh, parentFace, parentEdge->GetOrgVertex());
00639         } else {
00640             RefineFaceAtVertex(mesh, parentFace, parentEdge->GetDestVertex());
00641         }
00642 
00643         // It should always be the case that the opposite now exists -
00644         // we can't have a boundary case here
00645         assert(edge->GetOpposite());
00646     } else {
00647         HbrVertex<T>* parentVertex = edge->GetOrgVertex()->GetParentVertex();
00648         parentEdge = edge->GetDestVertex()->GetParentEdge();
00649         if (!parentVertex) {
00650             parentVertex = edge->GetDestVertex()->GetParentVertex();
00651             parentEdge = edge->GetOrgVertex()->GetParentEdge();
00652         }
00653 
00654         if (parentVertex) {
00655 
00656             assert(parentEdge);
00657 
00658 #ifdef HBR_DEBUG
00659             std::cerr << "\nparent edge is " << *parentEdge << "\n";
00660 #endif
00661 
00662             // 1. Go up to the parent of my face
00663 
00664             parentFace = edge->GetFace()->GetParent();
00665 #ifdef HBR_DEBUG
00666             std::cerr << "\nparent face is " << *parentFace << "\n";
00667 #endif
00668 
00669             // 2. Ask the opposite face (if it exists) to refine
00670             if (parentFace) {
00671 
00672                 // A vertex can be associated with either of two
00673                 // parent halfedges. If the parent edge that we're
00674                 // interested in doesn't match then we should look at
00675                 // its opposite
00676                 if (parentEdge->GetFace() != parentFace)
00677                     parentEdge = parentEdge->GetOpposite();
00678                 assert(parentEdge->GetFace() == parentFace);
00679 
00680                 // Make sure the parent edge has its neighbor as well
00681                 GuaranteeNeighbor(mesh, parentEdge);
00682 
00683                 // Now access that neighbor and refine it
00684                 if (parentEdge->GetRightFace()) {
00685                     RefineFaceAtVertex(mesh, parentEdge->GetRightFace(), parentVertex);
00686 
00687                     // FIXME: assertion?
00688                     assert(edge->GetOpposite());
00689                 }
00690             }

00695 template <class T>
00696 void
00697 HbrBilinearSubdivision<T>::GuaranteeNeighbors(HbrMesh<T>* mesh, HbrVertex<T>* vertex) {
00698 
00699 #ifdef HBR_DEBUG
00700     std::cerr << "\n\nneighbor guarantee at " << *vertex << " invoked\n";
00701 #endif
00702 
00703     // If the vertex is a child of a face, guaranteeing the neighbors
00704     // of the vertex is simply a matter of ensuring the parent face
00705     // has refined.
00706     HbrFace<T>* parentFace = vertex->GetParentFace();
00707     if (parentFace) {
00708 
00709 #ifdef HBR_DEBUG
00710         std::cerr << "  forcing full refine on parent face\n";
00711 #endif
00712         Refine(mesh, parentFace);
00713         return;
00714     }
00715 
00716     // Otherwise if the vertex is a child of an edge, we need to
00717     // ensure that the parent faces on either side of the parent edge
00718     // 1) exist, and 2) have refined at both vertices of the parent
00719     // edge
00720     HbrHalfedge<T>* parentEdge = vertex->GetParentEdge();
00721     if (parentEdge) {
00722 
00723 #ifdef HBR_DEBUG
00724         std::cerr << "  forcing full refine on adjacent faces of parent edge\n";
00725 #endif
00726         HbrVertex<T>* dest = parentEdge->GetDestVertex();
00727         HbrVertex<T>* org = parentEdge->GetOrgVertex();
00728         GuaranteeNeighbor(mesh, parentEdge);
00729         parentFace = parentEdge->GetLeftFace();
00730         RefineFaceAtVertex(mesh, parentFace, dest);
00731         RefineFaceAtVertex(mesh, parentFace, org);
00732 
00733 #ifdef HBR_DEBUG
00734         std::cerr << "    on the right face?\n";
00735 #endif
00736         parentFace = parentEdge->GetRightFace();
00737         // The right face may not necessarily exist even after
00738         // GuaranteeNeighbor
00739         if (parentFace) {
00740             RefineFaceAtVertex(mesh, parentFace, dest);
00741             RefineFaceAtVertex(mesh, parentFace, org);
00742         }
00743 #ifdef HBR_DEBUG
00744         std::cerr << "  end force\n";
00745 #endif
00746         return;
00747     }
00748 
00749     // The last case: the vertex is a child of a vertex. In this case
00750     // we have to first recursively guarantee that the parent's
00751     // adjacent faces also exist.
00752     HbrVertex<T>* parentVertex = vertex->GetParentVertex();
00753     if (parentVertex) {
00754 
00755 #ifdef HBR_DEBUG
00756         std::cerr << "  recursive parent vertex guarantee call\n";
00757 #endif
00758         parentVertex->GuaranteeNeighbors();
00759 
00760         // And then we refine all the face neighbors of the
00761         // parentVertex
00762         HbrHalfedge<T>* start = parentVertex->GetIncidentEdge(), *edge;
00763         edge = start;
00764         while (edge) {
00765             HbrFace<T>* f = edge->GetLeftFace();
00766             RefineFaceAtVertex(mesh, f, parentVertex);
00767             edge = parentVertex->GetNextEdge(edge);
00768             if (edge == start) break;
00769         }
00770     }
00771 }
