00513 // Determines whether an edge is infinitely sharp as far as its
00514 // facevarying data is concerned. Happens if the faces on both sides
00515 // disagree on the facevarying data at either of the shared vertices
00516 // on the edge.
00517 template <class T>
00518 bool
00519 HbrHalfedge<T>::GetFVarInfiniteSharp(int datum) {
00520 
00521     // Check to see if already initialized
00522     int intindex = datum >> 4;
00523     int shift = (datum & 15) << 1;
00524     unsigned int *fvarinfsharp = getFVarInfSharp();
00525     unsigned int bits = (fvarinfsharp[intindex] >> shift) & 0x3;
00526     if (bits != 3) {
00527         assert (bits != 2);
00528         return bits ? true : false;
00529     }
00530 
00531     // If there is no face varying data it can't be infinitely sharp!
00532     const int fvarwidth = GetMesh()->GetTotalFVarWidth();
00533     if (!fvarwidth) {
00534         bits = ~(0x3 << shift);
00535         fvarinfsharp[intindex] &= bits;
00536         if (opposite) opposite->getFVarInfSharp()[intindex] &= bits;
00537         return false;
00538     }
00539 
00540     // If either incident face is missing, it's a geometric boundary
00541     // edge, and also a facevarying boundary edge
00542     HbrFace<T>* left = GetLeftFace(), *right = GetRightFace();
00543     if (!left || !right) {
00544         bits = ~(0x2 << shift);
00545         fvarinfsharp[intindex] &= bits;
00546         if (opposite) opposite->getFVarInfSharp()[intindex] &= bits;
00547         return true;
00548     }
00549 
00550     // Look for the indices on each face which correspond to the
00551     // origin and destination vertices of the edge
00552     int lorg = -1, ldst = -1, rorg = -1, rdst = -1, i, nv;
00553     HbrHalfedge<T>* e;
00554     e = left->GetFirstEdge();
00555     nv = left->GetNumVertices();
00556     for (i = 0; i < nv; ++i) {
00557         if (e->GetOrgVertex() == GetOrgVertex()) lorg = i;
00558         if (e->GetOrgVertex() == GetDestVertex()) ldst = i;
00559         e = e->GetNext();
00560     }
00561     e = right->GetFirstEdge();
00562     nv = right->GetNumVertices();
00563     for (i = 0; i < nv; ++i) {
00564         if (e->GetOrgVertex() == GetOrgVertex()) rorg = i;
00565         if (e->GetOrgVertex() == GetDestVertex()) rdst = i;
00566         e = e->GetNext();
00567     }
00568     assert(lorg >= 0 && ldst >= 0 && rorg >= 0 && rdst >= 0);
00569     // Compare the facevarying data to some tolerance
00570     const int startindex = GetMesh()->GetFVarIndices()[datum];
00571     const int width = GetMesh()->GetFVarWidths()[datum];
00572     if (!right->GetFVarData(rorg).Compare(left->GetFVarData(lorg), startindex, width, 0.001f) ||
00573         !right->GetFVarData(rdst).Compare(left->GetFVarData(ldst), startindex, width, 0.001f)) {
00574         bits = ~(0x2 << shift);
00575         fvarinfsharp[intindex] &= bits;
00576         if (opposite) opposite->getFVarInfSharp()[intindex] &= bits;
00577         return true;
00578     }
00579 
00580     bits = ~(0x3 << shift);
00581     fvarinfsharp[intindex] &= bits;
00582     if (opposite) opposite->getFVarInfSharp()[intindex] &= bits;
00583     return false;
00584 }
