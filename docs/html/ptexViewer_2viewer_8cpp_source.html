00157 cross(float *n, const float *p0, const float *p1, const float *p2) {
00158 
00159     float a[3] = { p1[0]-p0[0], p1[1]-p0[1], p1[2]-p0[2] };
00160     float b[3] = { p2[0]-p0[0], p2[1]-p0[1], p2[2]-p0[2] };
00161     n[0] = a[1]*b[2]-a[2]*b[1];
00162     n[1] = a[2]*b[0]-a[0]*b[2];
00163     n[2] = a[0]*b[1]-a[1]*b[0];
00164 
00165     float rn = 1.0f/sqrtf(n[0]*n[0] + n[1]*n[1] + n[2]*n[2]);
00166     n[0] *= rn;
00167     n[1] *= rn;
00168     n[2] *= rn;
00169 }

00783 void motion(int x, int y) {
00784 
00785     if (g_mbutton[0] && !g_mbutton[1] && !g_mbutton[2]) {
00786         // orbit

00790         // pan

00792         g_pan[1] += g_dolly*(y - g_prev_y)/g_height;
00793     } else if (g_mbutton[0] && g_mbutton[1] && !g_mbutton[2]) {
00794         // dolly
00795         g_dolly -= g_dolly*0.01f*(x - g_prev_x);
00796         if(g_dolly <= 0.01) g_dolly = 0.01f;
00797     }
00798

00800     g_prev_y = float(y);
