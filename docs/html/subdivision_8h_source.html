00239     float sharpness = edge->GetSharpness();
00240 
00241     // In all methods, if the parent edge is infinitely sharp, the
00242     // child edge is also infinitely sharp
00243     if (sharpness >= HbrHalfedge<T>::k_InfinitelySharp) {
00244         subedge->SetSharpness(HbrHalfedge<T>::k_InfinitelySharp);
00245     }
00246 
00247     // Chaikin's curve subdivision: use 3/4 of the parent sharpness,
00248     // plus 1/4 of crease sharpnesses incident to vertex
00249     else if (creaseSubdivision == HbrSubdivision<T>::k_CreaseChaikin) {
00250 
00251         float childsharp = 0.0f;
00252 
00253         // Add 1/4 of the sharpness of all crease edges incident to
00254         // the vertex (other than this crease edge)
00255         std::vector<HbrHalfedge<T>*> edges;
00256         vertex->GuaranteeNeighbors();
00257         vertex->GetSurroundingEdges(std::back_inserter(edges));
00258 
00259         int n = 0;
00260         for (typename std::vector<HbrHalfedge<T>*>::iterator ei = edges.begin(); ei != edges.end(); ++ei) {
00261             if (*ei == edge) continue;
00262             if ((*ei)->GetSharpness() > HbrHalfedge<T>::k_Smooth) {
00263                 childsharp += (*ei)->GetSharpness();
00264                 n++;
00265             }
00266         }
00267 
00268         if (n) {
00269             childsharp = childsharp * 0.25f / n;
00270         }
00271 
00272         // Add 3/4 of the sharpness of this crease edge
00273         childsharp += sharpness * 0.75f;
00274         childsharp -= 1.0f;
00275         if (childsharp < (float) HbrHalfedge<T>::k_Smooth) {
00276             childsharp = (float) HbrHalfedge<T>::k_Smooth;
00277         }
00278         subedge->SetSharpness(childsharp);
00279 
00280     } else {
00281         sharpness -= 1.0f;
00282         if (sharpness < (float) HbrHalfedge<T>::k_Smooth) {
00283             sharpness = (float) HbrHalfedge<T>::k_Smooth;
00284         }
00285         subedge->SetSharpness(sharpness);
