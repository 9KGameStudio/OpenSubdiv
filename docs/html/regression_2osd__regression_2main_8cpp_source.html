00170 // Returns true if a vertex or any of its parents is on a boundary
00171 bool VertexOnBoundary( xyzvertex const * v ) {
00172 
00173     if (not v)
00174         return false;
00175 
00176     if (v->OnBoundary())
00177         return true;
00178 
00179     xyzvertex const * pv = v->GetParentVertex();
00180     if (pv)
00181         return VertexOnBoundary(pv);
00182     else {
00183         xyzhalfedge const * pe = v->GetParentEdge();
00184         if (pe) {
00185               return VertexOnBoundary(pe->GetOrgVertex()) or
00186                      VertexOnBoundary(pe->GetDestVertex());
00187         } else {
00188             xyzface const * pf = v->GetParentFace(), * rootf = pf;
00189             while (pf) {
00190                 pf = pf->GetParent();
00191                 if (pf)
00192                     rootf=pf;
00193             }
00194             if (rootf)
00195                 for (int i=0; i<rootf->GetNumVertices(); ++i)
00196                     if (rootf->GetVertex(i)->OnBoundary())
00197                         return true;
00198         }
00199     }
00200     return false;
00201 }

00205 int checkVertexBuffer( xyzmesh * hmesh,
00206                        OpenSubdiv::OsdCpuVertexBuffer * vb,
00207                        std::vector<int> const & remap) {
00208     int count=0;
00209     float deltaAvg[3] = {0.0f, 0.0f, 0.0f},
00210           deltaCnt[3] = {0.0f, 0.0f, 0.0f};
00211 
00212     int nverts = hmesh->GetNumVertices();
00213     for (int i=0; i<nverts; ++i) {
00214 
00215         xyzvertex * hv = hmesh->GetVertex(i);
00216 
00217         float * ov = & vb->GetCpuBuffer()[ remap[ hv->GetID() ] * vb->GetNumElements() ];
00218 
00219         // boundary interpolation rules set to "none" produce "undefined" vertices on
00220         // boundary vertices : far does not match hbr for those, so skip comparison.
00221         if ( hmesh->GetInterpolateBoundaryMethod()==xyzmesh::k_InterpolateBoundaryNone and
00222              VertexOnBoundary(hv) )
00223              continue;
00224 
00225 
00226         if ( hv->GetData().GetPos()[0] != ov[0] )
00227             deltaCnt[0]++;
00228         if ( hv->GetData().GetPos()[1] != ov[1] )
00229             deltaCnt[1]++;
00230         if ( hv->GetData().GetPos()[2] != ov[2] )
00231             deltaCnt[2]++;
00232 
00233         float delta[3] = { hv->GetData().GetPos()[0] - ov[0],
00234                            hv->GetData().GetPos()[1] - ov[1],
00235                            hv->GetData().GetPos()[2] - ov[2] };
00236 
00237         deltaAvg[0]+=delta[0];
00238         deltaAvg[1]+=delta[1];
00239         deltaAvg[2]+=delta[2];
00240 
00241         float dist = sqrtf( delta[0]*delta[0]+delta[1]*delta[1]+delta[2]*delta[2]);
00242         if ( dist > PRECISION ) {
00243             printf("// HbrVertex<T> %d fails : dist=%.10f (%.10f %.10f %.10f)"
00244                    " (%.10f %.10f %.10f)\n", i, dist, hv->GetData().GetPos()[0],
00245                                                       hv->GetData().GetPos()[1],
00246                                                       hv->GetData().GetPos()[2],
00247                                                       ov[0],
00248                                                       ov[1],
00249                                                       ov[2] );
00250            count++;
00251         }
00252     }
00253 
00254     if (deltaCnt[0])
00255         deltaAvg[0]/=deltaCnt[0];
00256     if (deltaCnt[1])
00257         deltaAvg[1]/=deltaCnt[1];
00258     if (deltaCnt[2])
00259         deltaAvg[2]/=deltaCnt[2];
00260 
00261     printf("    delta ratio : (%d/%d %d/%d %d/%d)\n", (int)deltaCnt[0], nverts,
00262                                                       (int)deltaCnt[1], nverts,
00263                                                       (int)deltaCnt[2], nverts );
00264     printf("    average delta : (%.10f %.10f %.10f)\n", deltaAvg[0],
00265                                                         deltaAvg[1],
00266                                                         deltaAvg[2] );
00267     if (count==0)
00268         printf("  success !\n");
00269 
00270     return count;
00271 }
