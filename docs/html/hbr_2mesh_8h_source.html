00708     m_numCoarseFaces = 0;
00709     for (i = 0; i < nfaces; ++i) {
00710         if (faces[i]) {
00711             faces[i]->SetCoarse();

00717     GetVertices(std::back_inserter(vertexlist));
00718     for (typename std::vector<HbrVertex<T>*>::iterator vi = vertexlist.begin();
00719          vi != vertexlist.end(); ++vi) {
00720         HbrVertex<T>* vertex = *vi;
00721         if (vertex->IsConnected()) vertex->Finish();
00722     }
00723     // If interpolateboundary is on, process boundary edges
00724     if (interpboundarymethod == k_InterpolateBoundaryEdgeOnly || interpboundarymethod == k_InterpolateBoundaryEdgeAndCorner) {
00725         for (i = 0; i < nfaces; ++i) {
00726             if (HbrFace<T>* face = faces[i]) {
00727                 int nv = face->GetNumVertices();
00728                 for (int k = 0; k < nv; ++k) {
00729                     HbrHalfedge<T>* edge = face->GetEdge(k);
00730                     if (edge->IsBoundary()) {
00731                         edge->SetSharpness(HbrHalfedge<T>::k_InfinitelySharp);
00732                     }
00733                 }
00734             }
00735         }
00736     }
00737     // Process corners
00738     if (interpboundarymethod == k_InterpolateBoundaryEdgeAndCorner) {
00739         for (typename std::vector<HbrVertex<T>*>::iterator vi = vertexlist.begin();
00740              vi != vertexlist.end(); ++vi) {
00741             HbrVertex<T>* vertex = *vi;
00742             if (vertex && vertex->IsConnected() && vertex->OnBoundary() && vertex->GetCoarseValence() == 2) {
00743                 vertex->SetSharpness(HbrVertex<T>::k_InfinitelySharp);
00744             }
00745         }
00746     }
00747 
00748     // Sort the hierarchical edits
00749     if (!hierarchicalEdits.empty()) {
00750         HbrHierarchicalEditComparator<T> cmp;
00751         int nHierarchicalEdits = (int)hierarchicalEdits.size();
00752         std::sort(hierarchicalEdits.begin(), hierarchicalEdits.end(), cmp);
00753         // Push a sentinel null value - we rely upon this sentinel to
00754         // ensure face->GetHierarchicalEdits knows when to terminate
00755         hierarchicalEdits.push_back(0);
00756         j = 0;

00758         for (i = 0; i < nfaces; ++i) {
00759             if (faces[i]) {
00760                 while (j < nHierarchicalEdits && hierarchicalEdits[j]->GetFaceID() < i) {
00761                     ++j;
00762                 }
00763                 if (j < nHierarchicalEdits && hierarchicalEdits[j]->GetFaceID() == i) {
00764                     faces[i]->SetHierarchicalEdits(&hierarchicalEdits[j]);

00951 template <class T>
00952 void
00953 HbrMesh<T>::GarbageCollect() {
00954     if (gcVertices.empty()) return;
00955 
00956     static const size_t gcthreshold = 4096;
00957 
00958     if (gcVertices.size() <= gcthreshold) return;
00959     // Go through the list of garbage collectable vertices and gather
00960     // up the neighboring faces of those vertices which can be garbage
00961     // collected.
00962     std::vector<HbrFace<T>*> killlist;
00963     std::vector<HbrVertex<T>*> vlist;
00964 
00965     // Process the vertices in the same order as they were collected
00966     // (gcVertices used to be declared as a std::deque, but that was
00967     // causing unnecessary heap traffic).
00968     int numprocessed = (int)gcVertices.size() - gcthreshold / 2;
00969     for (int i = 0; i < numprocessed; ++i) {
00970         HbrVertex<T>* v = gcVertices[i];
00971         v->ClearCollected();
00972         if (v->IsUsed()) continue;
00973         vlist.push_back(v);
00974         HbrHalfedge<T>* start = v->GetIncidentEdge(), *edge;
00975         edge = start;
00976         while (edge) {
00977             HbrFace<T>* f = edge->GetLeftFace();
00978             if (!f->IsCollected()) {
00979                 f->SetCollected();
00980                 killlist.push_back(f);
00981             }
00982             edge = v->GetNextEdge(edge);
00983             if (edge == start) break;
00984         }
00985     }
00986 
00987     gcVertices.erase(gcVertices.begin(), gcVertices.begin() + numprocessed);
00988 
00989     // Delete those faces
00990     for (typename std::vector<HbrFace<T>*>::iterator fi = killlist.begin(); fi != killlist.end(); ++fi) {
00991         if ((*fi)->GarbageCollectable()) {
00992             DeleteFace(*fi);
00993         } else {
00994             (*fi)->ClearCollected();
00995         }
00996     }
00997 
00998     // Delete as many vertices as we can
00999     for (typename std::vector<HbrVertex<T>*>::iterator vi = vlist.begin(); vi != vlist.end(); ++vi) {
01000         HbrVertex<T>* v = *vi;
01001         if (!v->IsReferenced()) {
01002             DeleteVertex(v);
01003         }
01004     }
