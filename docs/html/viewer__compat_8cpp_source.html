00263 cross(float *n, const float *p0, const float *p1, const float *p2) {
00264 
00265     float a[3] = { p1[0]-p0[0], p1[1]-p0[1], p1[2]-p0[2] };
00266     float b[3] = { p2[0]-p0[0], p2[1]-p0[1], p2[2]-p0[2] };
00267     n[0] = a[1]*b[2]-a[2]*b[1];
00268     n[1] = a[2]*b[0]-a[0]*b[2];
00269     n[2] = a[0]*b[1]-a[1]*b[0];
00270 
00271     float rn = 1.0f/sqrtf(n[0]*n[0] + n[1]*n[1] + n[2]*n[2]);
00272     n[0] *= rn;
00273     n[1] *= rn;
00274     n[2] *= rn;
00275 }

00685         // pan

00689         // dolly

00691         if(g_dolly <= 0.01) g_dolly = 0.01f;
00692     }
00693

00695     g_prev_y = float(y);
