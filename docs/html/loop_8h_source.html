00603 void
00604 HbrLoopSubdivision<T>::GuaranteeNeighbor(HbrMesh<T>* mesh, HbrHalfedge<T>* edge) {
00605     if (edge->GetOpposite()) {
00606         return;
00607     }
00608 
00609 #ifdef HBR_DEBUG
00610     std::cerr << "\n\nneighbor guarantee at " << *edge << " invoked\n";
00611 #endif
00612 
00613     /*
00614       Imagine the following:
00615 
00616                       X
00617                      / \
00618                     /   \
00619                    /     \
00620                   X       \
00621                  /\        \
00622                2/  \3       \
00623                /    \        \
00624               X------X--------X
00625                  1
00626 
00627      If the parent of _both_ incident vertices are themselves edges,
00628      (like the edge marked 3 above), then this edge is in the center
00629      of the parent face. Refining the parent face in the middle or
00630      refining the parent face at one vertex (where the two parent
00631      edges meet) should suffice
00632     */
00633     HbrHalfedge<T>* parentEdge1 = edge->GetOrgVertex()->GetParentEdge();
00634     HbrHalfedge<T>* parentEdge2 = edge->GetDestVertex()->GetParentEdge();
00635     if (parentEdge1 && parentEdge2) {
00636 #ifdef HBR_DEBUG
00637         std::cerr << "two parent edge situation\n";
00638 #endif
00639         HbrFace<T>* parentFace = parentEdge1->GetFace();
00640         assert(parentFace == parentEdge2->GetFace());
00641         if(parentEdge1->GetOrgVertex() == parentEdge2->GetDestVertex()) {
00642             refineFaceAtMiddle(mesh, parentFace);
00643         } else {
00644             RefineFaceAtVertex(mesh, parentFace, parentEdge1->GetOrgVertex());
00645         }
00646         assert(edge->GetOpposite());
00647         return;
00648     }
00649 
00650     // Otherwise we're in the situation of edge 1 or edge 2 in the
00651     // diagram above.
00652     if (parentEdge1) {
00653 #ifdef HBR_DEBUG
00654         std::cerr << "parent edge 1 " << *parentEdge1 << "\n";
00655 #endif
00656         HbrVertex<T>* parentVertex2 = edge->GetDestVertex()->GetParentVertex();
00657         assert(parentVertex2);
00658         RefineFaceAtVertex(mesh, parentEdge1->GetLeftFace(), parentVertex2);
00659         if (parentEdge1->GetRightFace()) {
00660             RefineFaceAtVertex(mesh, parentEdge1->GetRightFace(), parentVertex2);
00661         }
00662     } else if (parentEdge2) {
00663 #ifdef HBR_DEBUG
00664         std::cerr << "parent edge 2 " << *parentEdge2 << "\n";
00665 #endif
00666         HbrVertex<T>* parentVertex1 = edge->GetOrgVertex()->GetParentVertex();
00667         assert(parentVertex1);
00668         RefineFaceAtVertex(mesh, parentEdge2->GetLeftFace(), parentVertex1);
00669         if (parentEdge2->GetRightFace()) {
00670             RefineFaceAtVertex(mesh, parentEdge2->GetRightFace(), parentVertex1);
00671         }
00672     }
00673 }

00675 template <class T>
00676 void
00677 HbrLoopSubdivision<T>::GuaranteeNeighbors(HbrMesh<T>* mesh, HbrVertex<T>* vertex) {
00678 
00679 #ifdef HBR_DEBUG
00680     std::cerr << "\n\nneighbor guarantee at " << *vertex << " invoked\n";
00681 #endif
00682 
00683     assert(vertex->GetParentFace() == 0);
00684 
00685     // The first case: the vertex is a child of an edge. Make sure
00686     // that the parent faces on either side of the parent edge exist,
00687     // and have 1) refined at both vertices of the parent edge, and 2)
00688     // have refined their "middle" face (which doesn't live at either
00689     // vertex).
00690 
00691     HbrHalfedge<T>* parentEdge = vertex->GetParentEdge();
00692     if (parentEdge) {
00693 #ifdef HBR_DEBUG
00694         std::cerr << "parent edge situation " << *parentEdge << "\n";
00695 #endif
00696         HbrVertex<T>* dest = parentEdge->GetDestVertex();
00697         HbrVertex<T>* org = parentEdge->GetOrgVertex();
00698         GuaranteeNeighbor(mesh, parentEdge);
00699         HbrFace<T>* parentFace = parentEdge->GetLeftFace();
00700         RefineFaceAtVertex(mesh, parentFace, dest);
00701         RefineFaceAtVertex(mesh, parentFace, org);
00702         refineFaceAtMiddle(mesh, parentFace);
00703         parentFace = parentEdge->GetRightFace();
00704         // The right face may not necessarily exist even after
00705         // GuaranteeNeighbor
00706         if (parentFace) {
00707             RefineFaceAtVertex(mesh, parentFace, dest);
00708             RefineFaceAtVertex(mesh, parentFace, org);
00709             refineFaceAtMiddle(mesh, parentFace);
00710         }
00711         return;
00712     }
00713 
00714     // The second case: the vertex is a child of a vertex. In this case
00715     // we have to recursively guarantee that the parent's adjacent
00716     // faces also exist.
00717     HbrVertex<T>* parentVertex = vertex->GetParentVertex();
00718     if (parentVertex) {
00719 #ifdef HBR_DEBUG
00720         std::cerr << "parent vertex situation " << *parentVertex << "\n";
00721 #endif
00722         parentVertex->GuaranteeNeighbors();
00723 
00724         // And then we refine all the face neighbors of the parent
00725         // vertex
00726         HbrHalfedge<T>* start = parentVertex->GetIncidentEdge(), *edge;
00727         edge = start;
00728         while (edge) {
00729             HbrFace<T>* f = edge->GetLeftFace();
00730             RefineFaceAtVertex(mesh, f, parentVertex);
00731             edge = parentVertex->GetNextEdge(edge);
00732             if (edge == start) break;
00733         }
00734     }
00735 }

00811 template <class T>
00812 HbrVertex<T>*
00813 HbrLoopSubdivision<T>::Subdivide(HbrMesh<T>* mesh, HbrHalfedge<T>* edge) {
00814 
00815 #ifdef HBR_DEBUG
00816     std::cerr << "Subdividing at " << *edge << "\n";
00817 #endif
00818     // Ensure the opposite face exists.
00819     GuaranteeNeighbor(mesh, edge);
00820 
00821     float esharp = edge->GetSharpness();
00822     HbrVertex<T>* v = mesh->NewVertex();
00823     T& data = v->GetData();
00824 
00825     // If there's the possibility of vertex edits on either vertex, we
00826     // have to make sure the edit has been applied
00827     if (mesh->HasVertexEdits()) {
00828         edge->GetOrgVertex()->GuaranteeNeighbors();
00829         edge->GetDestVertex()->GuaranteeNeighbors();
00830     }
00831 
00832     if (!edge->IsBoundary() && esharp <= 1.0f) {
00833 
00834         // Of the two half-edges, pick one of them consistently such
00835         // that the org and dest vertices are also consistent through
00836         // multi-threading. It doesn't matter as far as the
00837         // theoretical calculation is concerned, but it is desirable
00838         // to be consistent about it in the face of the limitations of
00839         // floating point commutativity. So we always pick the
00840         // half-edge such that its incident face is the smallest of
00841         // the two faces, as far as the face paths are concerned.
00842         if (edge->GetOpposite() && edge->GetOpposite()->GetFace()->GetPath() < edge->GetFace()->GetPath()) {
00843             edge = edge->GetOpposite();
00844         }
00845 
00846         // Handle both the smooth and fractional sharpness cases.  We
00847         // lerp between the sharp case (average of the two end points)
00848         // and the unsharp case (3/8 of each of the two end points
00849         // plus 1/8 of the two opposite face averages).
00850 
00851         // Lerp end point weight between non sharp contribution of
00852         // 3/8 and the sharp contribution of 0.5.
00853         float endPtWeight = 0.375f + esharp * (0.5f - 0.375f);
00854         data.AddWithWeight(edge->GetOrgVertex()->GetData(), endPtWeight);
00855         data.AddWithWeight(edge->GetDestVertex()->GetData(), endPtWeight);
00856 
00857         // Lerp the opposite pt weights between non sharp contribution
00858         // of 1/8 and the sharp contribution of 0.
00859         float oppPtWeight = 0.125f * (1 - esharp);
00860         HbrHalfedge<T>* ee = edge->GetNext();
00861         data.AddWithWeight(ee->GetDestVertex()->GetData(), oppPtWeight);
00862         ee = edge->GetOpposite()->GetNext();
00863         data.AddWithWeight(ee->GetDestVertex()->GetData(), oppPtWeight);
00864     } else {
00865         // Fully sharp edge, just average the two end points
00866         data.AddWithWeight(edge->GetOrgVertex()->GetData(), 0.5f);
00867         data.AddWithWeight(edge->GetDestVertex()->GetData(), 0.5f);
00868     }
00869 
00870     // Varying data is always the average of two end points
00871     data.AddVaryingWithWeight(edge->GetOrgVertex()->GetData(), 0.5f);
00872     data.AddVaryingWithWeight(edge->GetDestVertex()->GetData(), 0.5f);
00873 
00874 #ifdef HBR_DEBUG
00875     std::cerr << "  created " << *v << "\n";
00876 #endif
00877 
00878     // Only boundary edges will create extraordinary vertices
00879     if (edge->IsBoundary()) {
00880         v->SetExtraordinary();
00881     }
00882     return v;
00883 }

00885 template <class T>
00886 HbrVertex<T>*
00887 HbrLoopSubdivision<T>::Subdivide(HbrMesh<T>* mesh, HbrVertex<T>* vertex) {
00888 
00889     // Ensure the ring of faces around this vertex exists before
00890     // we compute the valence
00891     vertex->GuaranteeNeighbors();
00892 
00893     float valence = static_cast<float>(vertex->GetValence());
00894     float invvalence = 1.0f / valence;
00895 
00896     HbrVertex<T>* v = mesh->NewVertex();
00897     T& data = v->GetData();
00898 
00899     // Due to fractional weights we may need to do two subdivision
00900     // passes
00901     int masks[2];
00902     float weights[2];
00903     int passes;
00904     masks[0] = vertex->GetMask(false);
00905     masks[1] = vertex->GetMask(true);
00906     // If the masks are different, we subdivide twice: once using the
00907     // current mask, once using the mask at the next level of
00908     // subdivision, then use fractional mask weights to weigh
00909     // each weighing
00910     if (masks[0] != masks[1]) {
00911         weights[1] = vertex->GetFractionalMask();
00912         weights[0] = 1.0f - weights[1];
00913         passes = 2;
00914     } else {
00915         weights[0] = 1.0f;
00916         weights[1] = 0.0f;
00917         passes = 1;
00918     }
00919     for (int i = 0; i < passes; ++i) {
00920         switch (masks[i]) {
00921             case HbrVertex<T>::k_Smooth:
00922             case HbrVertex<T>::k_Dart: {
00923                 float beta = 0.25f * cosf((float)M_PI * 2.0f * invvalence) + 0.375f;
00924                 beta = beta * beta;
00925                 beta = (0.625f - beta) * invvalence;
00926 
00927                 data.AddWithWeight(vertex->GetData(), weights[i] * (1 - (beta * valence)));
00928 
00929                 HbrSubdivision<T>::AddSurroundingVerticesWithWeight(
00930                     mesh, vertex, weights[i] * beta, &data);
00931                 break;
00932             }
00933             case HbrVertex<T>::k_Crease: {
00934                 // Compute 3/4 of old vertex value
00935                 data.AddWithWeight(vertex->GetData(), weights[i] * 0.75f);
00936 
00937                 // Add 0.125f of the (hopefully only two!) neighbouring
00938                 // sharp edges
00939                 HbrSubdivision<T>::AddCreaseEdgesWithWeight(
00940                     mesh, vertex, i == 1, weights[i] * 0.125f, &data);
00941                 break;
00942             }
00943             case HbrVertex<T>::k_Corner:
00944             default: {
00945                 // Just copy the old value
00946                 data.AddWithWeight(vertex->GetData(), weights[i]);
00947                 break;
00948             }
00949         }
00950     }
00951 
00952     // Varying data is always just propagated down
00953     data.AddVaryingWithWeight(vertex->GetData(), 1.0f);
00954 
00955 #ifdef HBR_DEBUG
00956     std::cerr << "Subdividing at " << *vertex << "\n";
00957     std::cerr << "  created " << *v << "\n";
00958 #endif
00959     // Inherit extraordinary flag and sharpness
00960     if (vertex->IsExtraordinary()) v->SetExtraordinary();
00961     float sharp = vertex->GetSharpness();
00962     if (sharp >= HbrVertex<T>::k_InfinitelySharp) {
00963         v->SetSharpness(HbrVertex<T>::k_InfinitelySharp);
00964     } else if (sharp > HbrVertex<T>::k_Smooth) {
00965         v->SetSharpness(std::max((float) HbrVertex<T>::k_Smooth, sharp - 1.0f));
00966     } else {
00967         v->SetSharpness(HbrVertex<T>::k_Smooth);
00968     }
00969     return v;
00970 }
