00170 bool
00171 HbrHierarchicalEdit<T>::IsRelevantToFace(HbrFace<T>* face) const {
00172 
00173     // Key assumption: the face's first vertex edit is relevant to
00174     // that face. We will then compare ourselves to that edit and if
00175     // the first part of our subpath is identical to the entirety of
00176     // that subpath, this edit is relevant.
00177 
00178     // Calling code is responsible for making sure we don't
00179     // dereference a null pointer here
00180     HbrHierarchicalEdit<T>* p = *face->GetHierarchicalEdits();
00181     if (!p) return false;
00182 
00183     if (this == p) return true;
00184 
00185     if (faceid != p->faceid) return false;
00186 
00187     // If our path length is less than the face depth, it should mean
00188     // that we're dealing with another face somewhere up the path, so
00189     // we're not relevant
00190     if (nsubfaces < face->GetDepth()) return false;
00191 
00192     if (memcmp(subfaces, p->subfaces, face->GetDepth() * sizeof(unsigned char)) != 0) {
00193         return false;
00194     }
00195     return true;
00196 }
