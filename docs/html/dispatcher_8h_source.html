00072 // Compute dispatcher : allows client code to customize parts or the entire
00073 // computation process. This pattern aims at hiding the logic specific to
00074 // the subdivision algorithms and expose a simplified access to minimalistic
00075 // compute kernels. By default, meshes revert to a default dispatcher that
00076 // implements single-threaded CPU kernels.
00077 //
00078 // - derive a dispatcher class from this one
00079 // - override the virtual functions
00080 // - pass the derived dispatcher to the factory (one instance can be shared by many meshes)
00081 // - call the FarMesh::Subdivide() to trigger computations
00082 //
00083 // Note : the caller is responsible for deleting a custom dispatcher

00130 template <class T,class U> void
00131 FarDispatcher<T,U>::Refine( FarMesh<T,U> * mesh, int maxlevel, void * data) const {
00132 
00133     assert(mesh);
00134 
00135     FarSubdivisionTables<T,U> const * tables = mesh->GetSubdivision();
00136 
00137     FarVertexEditTables<T,U> const * edits = mesh->GetVertexEdit();
00138 
00139     if ( (maxlevel < 0) )
00140         maxlevel=tables->GetMaxLevel();
00141     else
00142         maxlevel = std::min(maxlevel, tables->GetMaxLevel());
00143 
00144     for (int i=1; i<maxlevel; ++i) {
00145         tables->Apply(i, data);
00146         if (edits)
00147             edits->Apply(i, data);
00148     }
